HTML elements

<html>


<!doctype html>
<html lang="en">
</html>

<title>


 <title>Ourapp | Home page</title>

<head>


<head>
    <meta charset="utf-8">
    <meta name="viewport"
                    content="width=device-width, initial-scale=1">
    <link rel="shortcut icon" href="/favicon.ico">
    <title>Ourapp | Home page</title>
    <link type="text/css" rel="stylesheet" href="style.css">
    <script type="text/javascript"
                    charset="utf-8" src="script.js"></script>
</head>

<body>


 <body>...All my page content goes here...</body>

        <h3>Second level sub heading</h3>

 <div class="style-me">..some styled grouped content...</div>

    ..some grouped phrasing content...
</span>

</p>
